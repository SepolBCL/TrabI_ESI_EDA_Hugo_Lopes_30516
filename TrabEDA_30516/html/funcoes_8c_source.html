130/*
131/ Função para ler uma matriz de um ficheiro e armazená-la em uma lista ligada
132/ Retorna o apontador para o início da matriz (lista ligada)
133*/
134Node* lerMatriz(const char* ficheiro, int* lins, int* cols, bool* r)
135{
136 Node* inicio = NULL;
137 FILE* fp = fopen(ficheiro, "r");
138 if (fp == NULL)
139 {
140 *r = false;
141 return NULL;
142 }
143 *r = true;
144 int num;
145 char delimitador; // Variável para o delimitador (espaço)
146 *lins = 1; // Inicializa o número de linhas
147 *cols = 0; // Inicializa o número de colunas
148
149 // Ciclo para ler os números do ficheiro e criar nós para cada elemento da matriz
150 while (fscanf(fp, "%d%c", &num, &delimitador) != EOF)
151 {
152 Node* novo = novoNo(num); // Chama a função para criar um novo número e aloca memória para o mesmo
153 if (novo == NULL)
154 return NULL;
155 inicio = InsereFim(inicio, novo); // Chama a função para inserir o novo número no final
156 (*cols)++;
157
158 // Verifica se chegou ao final da linha
159 if (delimitador == '\n')
160 {
161 (*lins)++; // Incrementa o número de linhas
162 *cols = 0; // Reinicia o número de colunas para a próxima linha
163 }
164 }
165
166 fclose(fp); // Fecha o ficheiro depois da leitura
167 return inicio; // Retorna o apontador para o início lista ligada (matriz)

203Node* mudarVal(Node* matriz, int lin, int col, int novonum)
